import { useState } from "react";
import { useAuth } from "@/contexts/AuthContext";
import { DashboardLayout } from "@/components/DashboardLayout";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Checkbox } from "@/components/ui/checkbox";
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from "@/components/ui/select";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { Badge } from "@/components/ui/badge";
import { toast } from "sonner";
import { MOCK_COURSES, SESSIONS, SEMESTERS } from "@/data/mockData";

const CourseRegistration = () => {
  const { student } = useAuth();

  // Pre-filled from student profile — read-only
  const faculty = student?.faculty || "";
  const department = student?.department || "";
  const level = String(student?.level || "");
  const session = SESSIONS[SESSIONS.length - 1]; // always use the latest session

  // Only semester is selectable by the student
  const [semester, setSemester] = useState<string>("");
  const [selected, setSelected] = useState<Set<string>>(new Set());

  const filteredCourses = MOCK_COURSES.filter(
    (c) =>
      (!faculty || c.faculty === faculty) &&
      (!department || c.department === department) &&
      (!level || c.level === Number(level)) &&
      (!semester || c.semester === Number(semester)) &&
      c.session === session
  );

  const toggleCourse = (id: string) => {
    setSelected((prev) => {
      const next = new Set(prev);
      next.has(id) ? next.delete(id) : next.add(id);
      return next;
    });
  };

  const totalUnits = filteredCourses.filter((c) => selected.has(c.id)).reduce((s, c) => s + c.units, 0);

  const handleRegister = () => {
    if (!semester) {
      toast.error("Please select a semester first");
      return;
    }
    if (selected.size === 0) {
      toast.error("Please select at least one course");
      return;
    }
    toast.success(`Successfully registered for ${selected.size} course(s) (${totalUnits} units)`);
    setSelected(new Set());
  };

  return (
    <DashboardLayout>
      <div className="space-y-6">
        <div>
          <h1 className="text-2xl font-bold">Course Registration</h1>
          <p className="text-muted-foreground">Select your courses for the semester</p>
        </div>

        {/* Student Info Summary (read-only) */}
        <Card>
          <CardHeader>
            <CardTitle className="text-base">Registration Details</CardTitle>
          </CardHeader>
          <CardContent>
            <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-5 gap-4">
              {/* Read-only fields */}
              <div className="space-y-1.5">
                <label className="text-sm font-medium text-muted-foreground">Faculty</label>
                <div className="flex h-10 w-full items-center rounded-md border border-input bg-muted px-3 py-2 text-sm text-muted-foreground">
                  {faculty || "—"}
                </div>
              </div>
              <div className="space-y-1.5">
                <label className="text-sm font-medium text-muted-foreground">Department</label>
                <div className="flex h-10 w-full items-center rounded-md border border-input bg-muted px-3 py-2 text-sm text-muted-foreground">
                  {department || "—"}
                </div>
              </div>
              <div className="space-y-1.5">
                <label className="text-sm font-medium text-muted-foreground">Level</label>
                <div className="flex h-10 w-full items-center rounded-md border border-input bg-muted px-3 py-2 text-sm text-muted-foreground">
                  {level ? `${level} Level` : "—"}
                </div>
              </div>
              <div className="space-y-1.5">
                <label className="text-sm font-medium text-muted-foreground">Session</label>
                <div className="flex h-10 w-full items-center rounded-md border border-input bg-muted px-3 py-2 text-sm text-muted-foreground">
                  {session}
                </div>
              </div>

              {/* Only editable field */}
              <div className="space-y-1.5">
                <label className="text-sm font-medium">Semester <span className="text-destructive">*</span></label>
                <Select value={semester} onValueChange={(v) => { setSemester(v); setSelected(new Set()); }}>
                  <SelectTrigger>
                    <SelectValue placeholder="Select semester" />
                  </SelectTrigger>
                  <SelectContent>
                    {SEMESTERS.map((s) => (
                      <SelectItem key={s} value={String(s)}>Semester {s}</SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>
            </div>
          </CardContent>
        </Card>

        {/* Course Table */}
        <Card>
          <CardHeader className="flex flex-col sm:flex-row sm:items-center justify-between gap-2">
            <CardTitle className="text-base">
              {semester
                ? `Available Courses — Semester ${semester} (${filteredCourses.length})`
                : "Available Courses"}
            </CardTitle>
            {semester && (
              <div className="text-sm text-muted-foreground">
                Selected:{" "}
                <span className="font-semibold text-foreground">{selected.size}</span> courses •{" "}
                <span className="font-semibold text-foreground">{totalUnits}</span> units
              </div>
            )}
          </CardHeader>
          <CardContent>
            {!semester ? (
              <p className="text-center text-muted-foreground py-8">
                Please select a semester above to view available courses.
              </p>
            ) : filteredCourses.length === 0 ? (
              <p className="text-center text-muted-foreground py-8">
                No courses available for your department in Semester {semester}, {session}.
              </p>
            ) : (
              <Table>
                <TableHeader>
                  <TableRow>
                    <TableHead className="w-10"></TableHead>
                    <TableHead>Code</TableHead>
                    <TableHead>Title</TableHead>
                    <TableHead className="text-center">Units</TableHead>
                    <TableHead className="text-center">Type</TableHead>
                  </TableRow>
                </TableHeader>
                <TableBody>
                  {filteredCourses.map((course) => (
                    <TableRow
                      key={course.id}
                      className="cursor-pointer"
                      onClick={() => toggleCourse(course.id)}
                    >
                      <TableCell>
                        <Checkbox
                          checked={selected.has(course.id)}
                          onCheckedChange={() => toggleCourse(course.id)}
                        />
                      </TableCell>
                      <TableCell className="font-medium">{course.code}</TableCell>
                      <TableCell>{course.title}</TableCell>
                      <TableCell className="text-center">{course.units}</TableCell>
                      <TableCell className="text-center">
                        <Badge variant={course.type === "compulsory" ? "default" : "secondary"}>
                          {course.type}
                        </Badge>
                      </TableCell>
                    </TableRow>
                  ))}
                </TableBody>
              </Table>
            )}
            {semester && (
              <div className="flex justify-end mt-4">
                <Button onClick={handleRegister} disabled={selected.size === 0}>
                  Register Selected Courses
                </Button>
              </div>
            )}
          </CardContent>
        </Card>
      </div>
    </DashboardLayout>
  );
};

export default CourseRegistration;
